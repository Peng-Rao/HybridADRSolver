190 void precompute_coefficient_data() {
191 Assert(problem_ptr != nullptr, ExcNotInitialized());
192
193 const unsigned int n_cells = this->data->n_cell_batches();
194 const unsigned int n_q_points = Utilities::pow(fe_degree + 1, dim);
195
196 diffusion_coefficients.resize(n_cells);
197 advection_coefficients.resize(n_cells);
198 reaction_coefficients.resize(n_cells);
199
200 // Use FEEvaluation to get quadrature points
201 FEEvaluation<dim, fe_degree, fe_degree + 1, 1, Number> phi(*this->data);
202
203 for (unsigned int cell = 0; cell < n_cells; ++cell) {
204 phi.reinit(cell);
205
206 diffusion_coefficients[cell].resize(n_q_points);
207 advection_coefficients[cell].resize(n_q_points);
208 reaction_coefficients[cell].resize(n_q_points);
209
210 for (unsigned int q = 0; q < n_q_points; ++q) {
211 // Get the vectorized quadrature point
212 Point<dim, VectorizedArray<Number>> q_point =
213 phi.quadrature_point(q);
214
215 // Evaluate coefficients for each lane in the vector
216 VectorizedArray<Number> mu;
217 Tensor<1, dim, VectorizedArray<Number>> beta;
218 VectorizedArray<Number> gamma;
219
220 for (unsigned int v = 0; v < VectorizedArray<Number>::size();
221 ++v) {
222 Point<dim> p;
223 for (unsigned int d = 0; d < dim; ++d)
224 p[d] = q_point[d][v];
225
226 mu[v] = problem_ptr->diffusion_coefficient(p);
227 gamma[v] = problem_ptr->reaction_coefficient(p);
228
229 Tensor<1, dim> beta_scalar =
230 problem_ptr->advection_field(p);
231 for (unsigned int d = 0; d < dim; ++d)
232 beta[d][v] = beta_scalar[d];
233 }
234
235 diffusion_coefficients[cell][q] = mu;
236 advection_coefficients[cell][q] = beta;
237 reaction_coefficients[cell][q] = gamma;
238 }
239 }

248 void
249 local_apply(const MatrixFree<dim, Number>& mf_data, VectorType& dst,
250 const VectorType& src,
251 const std::pair<unsigned int, unsigned int>& cell_range) const {
252 FEEvaluation<dim, fe_degree, fe_degree + 1, 1, Number> phi(mf_data);
253
254 for (unsigned int cell = cell_range.first; cell < cell_range.second;
255 ++cell) {
256 phi.reinit(cell);
257 phi.gather_evaluate(src, EvaluationFlags::values |
258 EvaluationFlags::gradients);
259
260 for (unsigned int q = 0; q < phi.n_q_points; ++q) {
261 const VectorizedArray<Number> u_val = phi.get_value(q);
262 const Tensor<1, dim, VectorizedArray<Number>> grad_u =
263 phi.get_gradient(q);
264
265 // Diffusion: mu * grad(u)
266 Tensor<1, dim, VectorizedArray<Number>> flux =
267 diffusion_coefficients[cell][q] * grad_u;
268
269 // Advection contribution to flux: (beta · grad(u)) * v
270 // This is handled via the value term below
271 VectorizedArray<Number> advection_val =
272 advection_coefficients[cell][q] * grad_u;
273
274 // Reaction: gamma * u
275 VectorizedArray<Number> val =
276 reaction_coefficients[cell][q] * u_val + advection_val;
277
278 phi.submit_gradient(flux, q);
279 phi.submit_value(val, q);
280 }
281
282 phi.integrate_scatter(
283 EvaluationFlags::values | EvaluationFlags::gradients, dst);
284 }

292 void local_apply_transpose(
293 const MatrixFree<dim, Number>& mf_data, VectorType& dst,
294 const VectorType& src,
295 const std::pair<unsigned int, unsigned int>& cell_range) const {
296 FEEvaluation<dim, fe_degree, fe_degree + 1, 1, Number> phi(mf_data);
297
298 for (unsigned int cell = cell_range.first; cell < cell_range.second;
299 ++cell) {
300 phi.reinit(cell);
301 phi.gather_evaluate(src, EvaluationFlags::values |
302 EvaluationFlags::gradients);
303
304 for (unsigned int q = 0; q < phi.n_q_points; ++q) {
305 const VectorizedArray<Number> u_val = phi.get_value(q);
306 const Tensor<1, dim, VectorizedArray<Number>> grad_u =
307 phi.get_gradient(q);
308
309 // Diffusion (symmetric)
310 Tensor<1, dim, VectorizedArray<Number>> flux =
311 diffusion_coefficients[cell][q] * grad_u;
312
313 // Transpose of advection: -(beta · grad(v)) * u
314 // In weak form, this becomes +(div(beta) * u, v) - (beta * u,
315 // grad(v)) For divergence-free beta, this simplifies
316 VectorizedArray<Number> advection_val =
317 -advection_coefficients[cell][q] * grad_u;
318
319 // Reaction (symmetric)
320 VectorizedArray<Number> val =
321 reaction_coefficients[cell][q] * u_val + advection_val;
322
323 phi.submit_gradient(flux, q);
324 phi.submit_value(val, q);
325 }
326
327 phi.integrate_scatter(
328 EvaluationFlags::values | EvaluationFlags::gradients, dst);
329 }

335 void local_compute_diagonal(
336 const MatrixFree<dim, Number>& mf_data, VectorType& dst,
337 const VectorType& /*src*/,
338 const std::pair<unsigned int, unsigned int>& cell_range) const {
339 FEEvaluation<dim, fe_degree, fe_degree + 1, 1, Number> phi(mf_data);
340
341 AlignedVector<VectorizedArray<Number>> diagonal_values(
342 phi.dofs_per_cell);
343
344 for (unsigned int cell = cell_range.first; cell < cell_range.second;
345 ++cell) {
346 phi.reinit(cell);
347
348 for (unsigned int i = 0; i < phi.dofs_per_cell; ++i) {
349 // Set unit vector
350 for (unsigned int j = 0; j < phi.dofs_per_cell; ++j)
351 phi.submit_dof_value(VectorizedArray<Number>(), j);
352 phi.submit_dof_value(make_vectorized_array<Number>(1.0), i);
353
354 phi.evaluate(EvaluationFlags::values |
355 EvaluationFlags::gradients);
356
357 for (unsigned int q = 0; q < phi.n_q_points; ++q) {
358 const VectorizedArray<Number> u_val = phi.get_value(q);
359 const Tensor<1, dim, VectorizedArray<Number>> grad_u =
360 phi.get_gradient(q);
361
362 Tensor<1, dim, VectorizedArray<Number>> flux =
363 diffusion_coefficients[cell][q] * grad_u;
364
365 VectorizedArray<Number> advection_val =
366 advection_coefficients[cell][q] * grad_u;
367
368 VectorizedArray<Number> val =
369 reaction_coefficients[cell][q] * u_val + advection_val;
370
371 phi.submit_gradient(flux, q);
372 phi.submit_value(val, q);
373 }
374
375 phi.integrate(EvaluationFlags::values |
376 EvaluationFlags::gradients);
377 diagonal_values[i] = phi.get_dof_value(i);
378 }
379
380 for (unsigned int i = 0; i < phi.dofs_per_cell; ++i)
381 phi.submit_dof_value(diagonal_values[i], i);
382 phi.distribute_local_to_global(dst);
383 }
384 }
