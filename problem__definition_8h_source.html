111template <int dim>
112class ExactSolution : public Function<dim> {
113public:
118 double value(const Point<dim>& p, const unsigned int) const override {
119 double val = 1.0;
120 for (unsigned int d = 0; d < dim; ++d)
121 val *= std::sin(PI * p[d]);
122 return val;
123 }
124
128 Tensor<1, dim> gradient(const Point<dim>& p, const unsigned int) const override {
129 Tensor<1, dim> grad;
130 for (unsigned int d = 0; d < dim; ++d) {
131 grad[d] = PI * std::cos(PI * p[d]);
132 for (unsigned int other = 0; other < dim; ++other)
133 if (d != other) grad[d] *= std::sin(PI * p[other]);
134 }
135 return grad;
136 }
137};

145template <int dim>
146class NeumannFluxRight : public Function<dim> {
147public:
148 double value(const Point<dim>& p, const unsigned int) const override {
149 // Flux = du/dx at x=1.
150 // u = sin(pi*x) * sin(pi*y)...
151 // du/dx = pi * cos(pi*x) * sin(pi*y)...
152 // At x=1, cos(pi*x) = -1.
153 double val = -1.0 * PI;
154 for (unsigned int d = 1; d < dim; ++d)
155 val *= std::sin(PI * p[d]);
156 return val;
157 }
158};

168template <int dim>
169class ADRProblem : public ProblemInterface<dim> {
170public:
171 ADRProblem()
172 : zero_function(1), neumann_flux(), exact_solution() {}
173
175 double diffusion_coefficient(const Point<dim>&) const override { return 1.0; }
176
178 double reaction_coefficient(const Point<dim>&) const override { return 0.1; }
179
185 Tensor<1, dim> advection_field(const Point<dim>& p) const override {
186 Tensor<1, dim> beta;
187 beta[0] = -p[1];
188 beta[1] = p[0];
189 if (dim == 3) beta[2] = 0.1;
190 return beta;
191 }
192
200 double source_term(const Point<dim>& p) const override {
201 const double u = exact_solution.value(p, 0);
202 Tensor<1, dim> grad = exact_solution.gradient(p, 0);
203
204 const double mu = diffusion_coefficient(p);
205 const double gamma = reaction_coefficient(p);
206 Tensor<1, dim> beta = advection_field(p);
207
208 // Laplacian of u = product(sin(pi*x_i)) is -d * pi^2 * u
209 const double lap = -1.0 * dim * PI * PI * u;
210
211 return -mu * lap + beta * grad + gamma * u;
212 }
213
218 [[nodiscard]] std::set<types::boundary_id> get_dirichlet_ids() const override {
219 std::set<types::boundary_id> ids;
220 ids.insert(0); // Left
221 ids.insert(2); // Bottom
222 ids.insert(3); // Top
223 if(dim == 3) { ids.insert(4); ids.insert(5); }
224 return ids;
225 }
226
231 [[nodiscard]] std::set<types::boundary_id> get_neumann_ids() const override {
232 return {1};
233 }
234
236 const Function<dim>& get_dirichlet_function(types::boundary_id) const override {
237 return zero_function;
238 }
239
241 const Function<dim>& get_neumann_function(const types::boundary_id id) const override {
242 if (id == 1) return neumann_flux;
243 Assert(false, ExcMessage("Unknown Neumann ID requested"));
244 return zero_function;
245 }
246
247 [[nodiscard]] bool has_exact_solution() const override { return true; }
248 const Function<dim>& get_exact_solution() const override { return exact_solution; }
249
251 [[nodiscard]] bool is_symmetric() const override { return false; }
252
253 [[nodiscard]] std::string get_name() const override { return "Manufactured ADR (Dirichlet + Neumann)"; }
254
255private:
256 Functions::ZeroFunction<dim> zero_function;
257 NeumannFluxRight<dim> neumann_flux;
258 ExactSolution<dim> exact_solution;
259};
